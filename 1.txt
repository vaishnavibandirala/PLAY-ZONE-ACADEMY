  <script>
    // Smooth scroll helper
    function smoothScroll(sel){
      const el = document.querySelector(sel);
      if(!el) return;
      el.scrollIntoView({behavior:'smooth',block:'start'});
    }

    // Toggle card details on click (single-open behavior)
    const cards = document.querySelectorAll('.card');
    cards.forEach(card=>{
      card.addEventListener('click',()=>{
        const game = card.getAttribute('data-game');
        const id = 'details-' + game;
        // hide other details
        document.querySelectorAll('.details').forEach(d=>{ if(d.id !== id) d.classList.remove('show'); });
        // toggle this
        const details = document.getElementById(id);
        if(details) details.classList.toggle('show');
      });
    });

    // Simple booking form behaviour (demo-only)
    const form = document.getElementById('bookingForm');
    form.addEventListener('submit', (e)=>{
      e.preventDefault();
      const data = new FormData(form);
      const name = data.get('name');
      const phone = data.get('phone');
      const game = data.get('game');
      const datetime = data.get('datetime');
      // in production: send to server or use EmailJS / API. here we just show a message.
      const msg = document.getElementById('formMsg');
      msg.textContent = `Thanks ${name}! Booking request for ${game} on ${datetime || 'your chosen time'} received. We'll contact you at ${phone}.`;
      form.reset();
    });

    // Small accessibility: keyboard open/close details with Enter
    cards.forEach(c=>{ c.setAttribute('tabindex','0'); c.addEventListener('keydown', (e)=>{ if(e.key==='Enter') c.click(); }); });
  </script>
